JS - Assignment Operators

  1. =
  2. +=
  3. -=
  4. *=
  5. /=
  6. %=
console.clear();

var num1 = 15;
var num2 = 20;

// =
num1 = 20;

// +=
num1 += 10;
num1 += num2;
console.log(num1);

// -=
num1 -= 10;
num1 -= num2;
console.log(num1);

// *=
num1 *= 20;
console.log(num1);

// /=
num1 /= 1000;
console.log(num1);

// %=
num1 %= 10;
console.log(num1);
